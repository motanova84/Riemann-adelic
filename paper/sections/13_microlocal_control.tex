\subsection{The QCAL Phase Space and Hyperbolic Map}

Let $\mathcal{M} = \mathbb{R}_u \times [-\pi, \pi]_k$ with $u = \log x$. The hyperbolic map $\mathcal{T}_F: \mathcal{M} \to \mathcal{M}$ is:

\[
\mathcal{T}_F(u, k) = \bigl(F(u),\; \alpha k + \epsilon \sin(u) \bigr)
\]

where $F: \mathbb{R} \to \mathbb{R}$ is differentiable with $\inf_{u \in \mathbb{R}} F(u) > 1$, $\alpha > 1$, and $\epsilon > 0$ the shear coupling constant.

The cotangent lift $d\mathcal{T}_F^*$ on $T^*\mathcal{M}$ with coordinates $(u, k; \xi_u, \xi_k)$ obeys:

\[
\begin{pmatrix} \xi_u' \\ \xi_k' \end{pmatrix} =
\begin{pmatrix} \frac{1}{F'(u)} & 0 \\ -\frac{\epsilon \cos(u)}{F'(u)} & \frac{1}{\alpha} \end{pmatrix}
\begin{pmatrix} \xi_u \\ \xi_k \end{pmatrix}
\]

\subsection{Modulated Stable Cone Field}

Define cone apertures modulated by the spatial coordinate:

\[
\chi_e(u) = \chi_a(u) = \chi_0(1 + \cos^2(u)), \quad \chi_0 > 0
\]

The narrow and wide stable cones are:

\[
C^s_e(u) = \{ |\xi_u| \le \chi_e(u)^{-1}|\xi_k| \}, \qquad
C^s_a(u) = \{ |\xi_k| \ge \chi_a(u)|\xi_u| \}
\]

\textbf{Theorem 2.1 (Strict Hyperbolic Invariance).} For all $(u,k) \in \mathcal{M}$:

\[
d\mathcal{T}_F^*\bigl(C^s_e(u)\bigr) \subset C^s_a(F(u))
\]

with uniform angular margin $\delta_0 = \frac{1+\beta}{2\alpha^2} - \frac{1}{2} > 0$ for $\beta = 1$, $\alpha^2 > 3/2$.

\subsection{Anisotropic Banach Space $\mathcal{H}_{W,V}(\mathcal{M})$}

The microlocal escape weight:

\[
W(u, k; \xi_u, \xi_k) = -m\ln(\alpha) \cdot \psi_u(\xi), \quad \psi_u \in C_c^\infty,\ \operatorname{supp}\psi_u \subset C^s_a(u)
\]

The spatial control potential:

\[
V(u) = \ln(1 + u^2), \qquad e^{-V(u)} = \frac{1}{1+u^2}
\]

The anisotropic Sobolev space:

\[
\mathcal{H}_{W,V}(\mathcal{M}) = \left\{ \phi \in \mathcal{S}'(\mathcal{M}) \;:\; \| e^{V} \cdot \operatorname{Op}(e^W)\phi \|_{L^2} < \infty \right\}
\]

\subsection{Controlled Ruelle Operator and Essential Spectrum Extinction}

\textbf{Definition.} The controlled Ruelle operator is:

\[
\mathcal{L}_{s,V}[\phi](u, k) = \frac{1}{1+u^2} \cdot \sum_{\mathcal{T}_F(u',k') = (u,k)} 
\frac{e^{s u'}}{| \det d\mathcal{T}_F(u',k') |^{1/2}} \, \phi(u',k')
\]

The regularized conjugate $\tilde{\mathcal{L}}_{s,V} = \operatorname{Op}(e^W) \circ \mathcal{L}_{s,V} \circ \operatorname{Op}(e^{-W})$ acts on $L^2(\mathcal{M})$.

\textbf{Theorem 5.1 (Essential Spectrum Extinction).}
\[
\| \tilde{\mathcal{L}}_{s,V} \|_{\text{ess}} = 0
\]

\textit{Proof sketch.} Construct $K_N$ as the quantization of a smooth cutoff $\chi_N$ with support in $B_{R_N} \subset T^*\mathcal{M}$, $R_N \sim \sqrt{N}$. The residual symbol $(1-\chi_N)\sigma_{\text{pr}}(\tilde{\mathcal{L}}_{s,V})$ has derivatives decaying as $\mathcal{O}(|u|^{-2})$ (spatial) and $\mathcal{O}(\|\xi\|^{-2m})$ (frequency). By the generalized Calderón-Vaillancourt theorem, $\|\mathcal{R}_N\| \le C \cdot \max(R_N^{-2}, R_N^{-2m}) \to 0$. Singular values decay sub-exponentially $s_j = \mathcal{O}(e^{-c j^{1/2}})$, hence $\tilde{\mathcal{L}}_{s,V}$ is trace class. ∎

\subsection{Fredholm Determinant and Preservation of Zeros}

\[
\det(I - \mathcal{L}_{s,V}) = \exp\left(-\sum_{m=1}^\infty \frac{1}{m} \sum_{p} \frac{p^{-m(s+1/2)}}{(1 + \log^2 p)^m}\right)
